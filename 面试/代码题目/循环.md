# 最长公共前缀

输入：strs = ["flower","flow","flight"]
输出："fl"

```js
function getMaxLength(strs) {
  if (!strs) {
    return "";
  }
  for (let i = 0, len = strs[0]?.length; i < len; i++) {
    for (let j = 1, jen = strs.length; j < jen; j++) {
      if (strs[0][i] !== strs[j][i]) {
        return strs[0].substring(0, i);
      }
    }
  }
  return strs[0];
}
// test
const strs = ["flower", "flow", "flight"];
getMaxLength(strs);
```

var longestCommonPrefix = function(strs) {
  if (!strs?.length){

}

# 笛卡尔积

给定输入：

```
[
  [1, 2],
  ['a', 'b']
]
```

输出

```
[
  [1, 'a'],
  [1, 'b'],
  [2, 'a'],
  [2, 'b']
]
```

### 暴力解法

```js
function cartesianProductIterative(arrays) {
  if (!arrays.length) return [];

  let result = [[]];

  for (const group of arrays) {
    const temp = [];

    for (const res of result) {
      for (const item of group) {
        temp.push([...res, item]);
      }
    }

    result = temp;
  }

  return result.map((arr) => arr.join(""));
}
```
