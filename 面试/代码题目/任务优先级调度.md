这个实现提供了一个基于依赖关系的任务调度系统，支持异步任务执行。你可以通过 addTask 方法添加任务，然后调用 run 方法启动调度执行。系统会自动处理任务依赖关系，确保任务按正确顺序执行。

```
class TaskScheduler {
  constructor() {
    this.tasks = new Map();
    this.executionQueue = [];
    this.completedTasks = new Set();
  }

  addTask(task) {
    if (this.tasks.has(task.id)) {
      throw new Error(`Task with id ${task.id} already exists`);
    }
    task.status = 'pending';
    this.tasks.set(task.id, task);
  }

  schedule() {
    // 重置执行队列
    this.executionQueue = [];

    // 找出所有没有依赖或依赖已完成的就绪任务
    for (const [id, task] of this.tasks) {
      if (task.status === 'pending') {
        const isReady = task.dependencies.every(depId =>
          this.completedTasks.has(depId)
        );

        if (isReady) {
          task.status = 'ready';
          this.executionQueue.push(id);
        }
      }
    }
  }

  async execute() {
    while (this.executionQueue.length > 0) {
      const taskId = this.executionQueue.shift();
      const task = this.tasks.get(taskId);

      try {
        task.status = 'running';
        await task.execute();
        task.status = 'completed';
        this.completedTasks.add(taskId);

        // 每次任务完成后重新调度
        this.schedule();
      } catch (error) {
        console.error(`Task ${taskId} failed:`, error);
        task.status = 'failed';
        break;
      }
    }
  }

  async run() {
    this.schedule();
    await this.execute();
  }
}

// 示例用法
const scheduler = new TaskScheduler();

scheduler.addTask({
  id: 'task1',
  name: 'Task 1',
  dependencies: [],
  execute: () => new Promise(resolve => {
    setTimeout(() => {
      console.log('Task 1 completed');
      resolve();
    }, 1000);
  })
});

scheduler.addTask({
  id: 'task2',
  name: 'Task 2',
  dependencies: ['task1'],
  execute: () => new Promise(resolve => {
    setTimeout(() => {
      console.log('Task 2 completed');
      resolve();
    }, 500);
  })
});

scheduler.run();

```
